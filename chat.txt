    def process_scanners(self, data, execution_plan):
        exec_plan = copy.deepcopy(execution_plan)
        for _, plan in exec_plan.items():
            results = []
            if "EncryptionDetector" in plan:
                result = self.run_scanner(data, "EncryptionDetector", plan["EncryptionDetector"], **self.kwargs)
                results.append(result)
                plan.pop("EncryptionDetector")
                if result["response"]["is_protected"]:
                    LogUtil.log(LogType.TRANSACTION, LogLevel.INFO, "File is encrypted, skipping further execute for scanners: {}.".format(", ".join(plan.keys())))
                    return results
            for scanner_nm, param in plan.items():
                result = self.run_scanner(data, scanner_nm, param, **self.kwargs)
                results.append(result)
                data = result["sanitized_data"] or data
        return results
