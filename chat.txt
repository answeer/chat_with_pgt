import cv2
import numpy as np

def denoise_watermarked_document(image_path):
    # 读取文档图片
    image = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)

    # 步骤 1: 应用双边滤波去除噪声
    # 双边滤波器能保留边缘，同时对平滑区域进行降噪处理
    denoised = cv2.bilateralFilter(image, d=9, sigmaColor=75, sigmaSpace=75)

    # 步骤 2: 自适应阈值增强对比度，使文档清晰
    final = cv2.adaptiveThreshold(denoised, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2)

    # 步骤 3: 如果需要，进一步处理水印区域 (可以自定义水印位置的处理逻辑)

    return final

# 运行示例
image_path = 'watermarked_document.jpg'
denoised_image = denoise_watermarked_document(image_path)

# 显示降噪后的文档
cv2.imshow('Denoised Document', denoised_image)
cv2.imwrite('denoised_watermarked_document.png', denoised_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
