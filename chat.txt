CREATE OR REPLACE FUNCTION swoosh.get_io_sanitation_master_id_pk_seq(
	table_name bigint)
    RETURNS bigint
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
declare
seqid varchar;
date varchar;
BEGIN
SELECT into date REPLACE(TO_CHAR(NOW()::date, 'yyyy/mm/dd'),'/','');
select into seqid lpad((nextval('swoosh.io_sanitation_master_id_pk_seq')||''),7,'0');
return (table_name||date||seqid);
END;
$BODY$;
