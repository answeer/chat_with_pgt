def transform_coordinates(bounding_boxes, matrix):
    transformed_boxes = []
    
    for box in bounding_boxes:
        x_min, y_min, x_max, y_max = box
        
        # Create corner points in homogeneous coordinates
        top_left = np.array([x_min, y_min, 1]).reshape(-1, 1)
        bottom_right = np.array([x_max, y_max, 1]).reshape(-1, 1)

        # Apply affine transformation
        transformed_top_left = np.dot(matrix, top_left).flatten()
        transformed_bottom_right = np.dot(matrix, bottom_right).flatten()

        # Append the transformed box
        transformed_boxes.append((
            transformed_top_left[0], transformed_top_left[1],
            transformed_bottom_right[0], transformed_bottom_right[1]
        ))

    return transformed_boxes
