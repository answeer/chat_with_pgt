def show_scanner_config():
    """展示扫描器配置"""
    st.markdown("### 🔍 Default Scanners Configuration")
    
    for category, scanners in DEFAULT_SCANNERS.items():
        # 美化分类名称
        category_name = category.replace('_', ' ').title()
        
        with st.expander(f"📁 {category_name}", expanded=True):
            for scanner_name, params in scanners.items():
                st.markdown(f"<div class='scanner-name'>🛡️ {scanner_name}</div>", unsafe_allow_html=True)
                
                if params:
                    st.code(f"Parameters:\n{json.dumps(params, indent=4)}", language='json')
                else:
                    st.caption("No parameters required")
